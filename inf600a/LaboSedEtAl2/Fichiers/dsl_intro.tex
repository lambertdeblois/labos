\begin{frame}[fragile]
\frametitle{Avez-vous d\'ej\`a utilis\'e\ldots\ \AlertTitle{make}?}


\begin{columns}

\begin{column}{5cm}
\begin{block}{}
{\footnotesize
\begin{semiverbatim}
  \alert{$ cat Makefile}
  hello: hello.c
      gcc -o hello hello.c
  
  clean:
      rm -f hello hello.o
\end{semiverbatim}
}
\end{block}
\end{column}


\begin{column}{5cm}
\begin{block}{}
{\footnotesize
\begin{semiverbatim}
\alert{$ make}
gcc -o hello hello.c

\end{semiverbatim}
}
\end{block}
\end{column}

\end{columns}

\NOTE{Used, on Unix, to automate program build tasks, i.e., to compile
and link programs but also, more generally, to automate tasks.}

\end{frame}
